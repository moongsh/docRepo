#命令来查看可用版本
$docker search nginx 
#安装
$ docker pull nginx:latest
#运行 nginx 容器：
$docker run --rm --name nginx-test -p 8080:80 -d nginx
参数说明：
      --rm：容器终止运行后，自动删除容器文件
      --name nginx-test：容器名字
      -p: 端口进行映射，将本地 8080 端口映射到容器内部的 80 端口
      -d：容器启动后，在后台运行

停止服务 docker stop a57cacaabe1b
重启服务 $docker start a57cacaabe1b
              $docker restart 容器id
卸载服务 docker stop a57cacaabe1b   //停止
$docker ps -a //Exited
$docker rm a57cacaabe1b  //运作着的进程是无法rm的，要先stop

$docker exec -it a57cacaabe1b bash
$which nginx

#建目录用于存放nginx配置文件、证书文件
$mkdir /opt/docker/nginx/conf.d -p
$mkdir /opt/docker/nginx/conf.d -p
#进入容器内部，把配置文件拷贝到主机目录
$docker cp nginx:/etc/nginx/conf.d/default.conf /opt/docker/nginx/conf.d/default.conf
$docker cp nginx:/etc/nginx/nginx.conf /opt/docker/nginx/conf.d/nginx.conf
#删除容器，重新启动容器
$ docker run --rm -d -p 8081:80 --name nginx-test-web \
  -v /opt/nginx/www:/usr/share/nginx/html \
  -v /opt/nginx/conf/nginx.conf:/etc/nginx/nginx.conf \
  -v /opt/nginx/logs:/var/log/nginx \
  nginx
